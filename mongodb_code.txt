using MongoDB.Driver; using System; using System.Collections.Generic;
using MongoDB.Bson; using MongoDB.Bson.Serialization.Attributes;

namespace WebApplication18 { public partial class Default :
System.Web.UI.Page { MongoClient client = new
MongoClient(“mongodb://localhost:27017”); IMongoDatabase db;
IMongoCollection collection;

        protected void Page_Load(object sender, EventArgs e)
        {
            db = client.GetDatabase("StudentDB");
            collection = db.GetCollection<Student>("students");
        }

        protected void btnInsert_Click(object sender, EventArgs e)
        {
            Student s = new Student
            {
                RollNo = int.Parse(txtRoll.Text),
                Name = txtName.Text,
                Course = txtCourse.Text
            };

            collection.InsertOne(s);
            lblMsg.Text = "Inserted Successfully";
        }

        protected void btnSelect_Click(object sender, EventArgs e)
        {
            var list = collection.Find(_ => true).ToList();
            GridView1.DataSource = list;
            GridView1.DataBind();
        }

        protected void btnUpdate_Click(object sender, EventArgs e)
        {
            var filter = Builders<Student>.Filter.Eq("RollNo", int.Parse(txtRoll.Text));

            var update = Builders<Student>.Update
                .Set("Name", txtName.Text)
                .Set("Course", txtCourse.Text);

            collection.UpdateOne(filter, update);

            lblMsg.Text = "Updated Successfully";
        }

        protected void btnDelete_Click(object sender, EventArgs e)
        {
            var filter = Builders<Student>.Filter.Eq("RollNo", int.Parse(txtRoll.Text));

            collection.DeleteOne(filter);

            lblMsg.Text = "Deleted Successfully";
        }
    }

    public class Student
    {
        public ObjectId Id { get; set; }
        public int RollNo { get; set; }
        public string Name { get; set; }
        public string Course { get; set; }
    }

}
