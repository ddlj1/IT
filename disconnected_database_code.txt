using System; using System.Data; using System.Data.SqlClient;

namespace WebApplication18 { public partial class WebForm1 :
System.Web.UI.Page { string cs = @“Data Source=(localdb);Initial
Catalog=StudentDB;Integrated Security=True”;

        SqlConnection con;
        SqlDataAdapter da;
        DataSet ds;

        protected void Page_Load(object sender, EventArgs e)
        {
            con = new SqlConnection(cs);
            da = new SqlDataAdapter("select * from Student", con);
            ds = new DataSet();
            da.Fill(ds, "Student");
        }

        protected void btnSelect_Click(object sender, EventArgs e)
        {
            GridView1.DataSource = ds.Tables["Student"];
            GridView1.DataBind();
        }

        protected void btnInsert_Click(object sender, EventArgs e)
        {
            DataRow dr = ds.Tables["Student"].NewRow();
            dr["RollNo"] = txtRoll.Text;
            dr["Name"] = txtName.Text;

            ds.Tables["Student"].Rows.Add(dr);

            SqlCommandBuilder cb = new SqlCommandBuilder(da);
            da.Update(ds, "Student");

            lblMsg.Text = "Inserted";
        }

        protected void btnUpdate_Click(object sender, EventArgs e)
        {
            foreach (DataRow dr in ds.Tables["Student"].Rows)
            {
                if (dr["RollNo"].ToString() == txtRoll.Text)
                {
                    dr["Name"] = txtName.Text;
                }
            }

            SqlCommandBuilder cb = new SqlCommandBuilder(da);
            da.Update(ds, "Student");

            lblMsg.Text = "Updated";
        }

        protected void btnDelete_Click(object sender, EventArgs e)
        {
            foreach (DataRow dr in ds.Tables["Student"].Rows)
            {
                if (dr["RollNo"].ToString() == txtRoll.Text)
                {
                    dr.Delete();
                }
            }

            SqlCommandBuilder cb = new SqlCommandBuilder(da);
            da.Update(ds, "Student");

            lblMsg.Text = "Deleted";
        }
    }

}
